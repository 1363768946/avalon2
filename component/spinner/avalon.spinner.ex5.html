spinner 用onChange监听改变(当value值非法时onChange不会触发)

spinner值: {{spinnerVal}}

<!DOCTYPE html>
<html>
    
    <head>
        <title>spinner demo 5</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <link type="text/css" rel="stylesheet" href="../style/avalon.doc.css">
        <script src="../../avalon.js"></script>
        <script src="../style/highlight/shCore.js"></script>
        <script>
            avalon.library("oni", {
                $ready: function () {
                    avalon.log("控件已经构建完毕")
                }
            })
            require(["./component/spinner/avalon.spinner"], function () {
                var demoVM = avalon.define({
                    $id: "demo",
                    spinnerVal: "",
                    $opts: {
                        onChange: function(val){
                            demoVM.spinnerVal = val
                        }
                    }
                })
    
                avalon.scan()
            })
        </script>
    </head>
    
    <body ms-controller="demo">
        <fieldset>
            <legend>spinner 用onChange监听改变(当value值非法时onChange不会触发)</legend>
            <oni:spinner config="$opts"></oni:spinner>
            <p>spinner值: {{spinnerVal}}</p>
        </fieldset>
        <div ms-skip></div>
    </body>

</html>
